import { OpenAIEmbeddings } from "@langchain/openai";
import { config } from "@/config";

export class EmbeddingsService {
  private embeddingsModel: OpenAIEmbeddings;

  constructor() {
    this.embeddingsModel = new OpenAIEmbeddings({
      modelName: config.openai.embeddingModel,
      openAIApiKey: config.openai.apiKey,
    });
  }

  async generateEmbedding(text: string): Promise<number[]> {
    return this.embeddingsModel.embedQuery(text);
  }

  calculateCosineSimilarity(a: number[], b: number[]): number {
    if (a.length !== b.length) {
      throw new Error("Vectors must have the same length");
    }

    const dotProduct = a.reduce(
      (sum, ai, i) => sum + (ai ?? 0) * (b[i] ?? 0),
      0
    );
    const normA = Math.sqrt(a.reduce((sum, ai) => sum + (ai ?? 0) ** 2, 0));
    const normB = Math.sqrt(b.reduce((sum, bi) => sum + (bi ?? 0) ** 2, 0));

    return normA && normB ? dotProduct / (normA * normB) : 0;
  }

  calculateCentroid(embeddings: number[][]): number[] {
    if (embeddings.length === 0) return [];

    const dimensions = embeddings[0]?.length;
    const result = Array(dimensions).fill(0);

    embeddings.forEach((emb) =>
      emb.forEach((val, i) => {
        result[i] += val ?? 0;
      })
    );

    return result.map((sum) => sum / embeddings.length);
  }

  updateCentroidIncremental(
    currentCentroid: number[],
    currentCount: number,
    newEmbedding: number[]
  ): number[] {
    const newCount = currentCount + 1;

    return currentCentroid.map(
      (val, i) =>
        ((val ?? 0) * currentCount + (newEmbedding[i] ?? 0)) / newCount
    );
  }
}

  /**
   * Main clustering function that groups posts semantically across sources
   */
  async clusterPosts(posts: PostForClustering[]): Promise<ClusteringResult> {
    logger.info(
      `[Enhanced Clustering] Starting clustering of ${posts.length} posts`
    );

    // Step 1: Group posts that describe same problem/solution
    // const clusters = await this.groupPostsBySemantic(posts);

    // Pre-group by category
    const postsByCategory = new Map<number | null, PostForClustering[]>();
    posts.forEach((post) => {
      const catId = post.category_id;
      if (!postsByCategory.has(catId)) {
        postsByCategory.set(catId, []);
      }
      postsByCategory.get(catId)!.push(post);
    });

    // Cluster within each category
    const allClusters: PostForClustering[][] = [];
    for (const [categoryId, categoryPosts] of postsByCategory.entries()) {
      const categoryClusters = await this.groupPostsBySemantic(categoryPosts);
      allClusters.push(...categoryClusters);
    }

    // Step 2: Assign cluster metadata and classify as problem/solution
    const enhancedClusters = await this.enhanceClusters(allClusters);

    // Step 3: Merge highly similar clusters (>= 0.85 similarity)
    const mergedClusters = await this.mergeSimilarClusters(enhancedClusters);

    // Step 4: Filter out singletons and create final result
    const finalClusters = mergedClusters.filter(
      (cluster) => cluster.member_count >= 2
    );

    const clusteredPostIds = new Set(
      finalClusters.flatMap((cluster) => cluster.member_ids)
    );
    const unclustered = posts
      .filter((post) => !clusteredPostIds.has(post.id))
      .map((post) => post.id);

    logger.info(
      `[Enhanced Clustering] Created ${finalClusters.length} clusters, ${unclustered.length} unclustered`
    );

    return {
      clusters: finalClusters,
      unclustered,
    };
  }

  /**
   * Groups posts by semantic similarity using embeddings
   */
  private async groupPostsBySemantic(
    posts: PostForClustering[],
    options?: { groupThreshold?: number }
  ): Promise<PostForClustering[][]> {
    const groupThreshold = options?.groupThreshold ?? 0.72; // tune this
    const n = posts.length;
    if (n === 0) return [];

    // Optional: print matrix for debugging
    // this.printSimilarityMatrix(posts);

    // Build adjacency list
    const adj: number[][] = Array.from({ length: n }, () => []);

    for (let i = 0; i < n; i++) {
      for (let j = i + 1; j < n; j++) {
        const p1 = posts[i];
        const p2 = posts[j];

        if (!p1.embedding || !p2.embedding) continue;

        let sim = embeddingsService.calculateCosineSimilarity(
          p1.embedding as number[],
          p2.embedding as number[]
        );

        // mild classification adjustment: reduce sim slightly if classifications differ
        if (
          p1.classification &&
          p2.classification &&
          p1.classification !== p2.classification
        ) {
          sim -= 0.06; // smaller penalty than 0.15
        }

        // stricter for updates/features (avoid mixing product-update posts into problem clusters)
        const isUpdateFeature =
          (p1.classification &&
            ["update", "feature"].includes(p1.classification.toLowerCase())) ||
          (p2.classification &&
            ["update", "feature"].includes(p2.classification.toLowerCase()));
        const effectiveThreshold = isUpdateFeature
          ? Math.max(groupThreshold, 0.8)
          : groupThreshold;

        if (sim >= effectiveThreshold) {
          adj[i].push(j);
          adj[j].push(i);
        }
      }
    }

    // Extract connected components via BFS
    const visited = new Array<boolean>(n).fill(false);
    const clusters: PostForClustering[][] = [];

    for (let i = 0; i < n; i++) {
      if (visited[i]) continue;
      const queue = [i];
      visited[i] = true;
      const comp: PostForClustering[] = [];

      while (queue.length > 0) {
        const cur = queue.shift()!;
        comp.push(posts[cur]);
        for (const nb of adj[cur]) {
          if (!visited[nb]) {
            visited[nb] = true;
            queue.push(nb);
          }
        }
      }

      clusters.push(comp);
    }

    return clusters;
  }

  /**
   * Calculate semantic similarity between two posts
   */
  private async calculateSimilarity(
    post1: PostForClustering,
    post2: PostForClustering
  ): Promise<number> {
    // Penalize different classifications
    const classificationPenalty =
      (post1.classification === "update" ||
        post2.classification === "update") &&
      post1.classification !== post2.classification
        ? 0.15
        : 0;

    if (post1.embedding && post2.embedding) {
      const similarity = embeddingsService.calculateCosineSimilarity(
        post1.embedding as number[],
        post2.embedding as number[]
      );
      return Math.max(0, similarity - classificationPenalty);
    }

    // Fallback to keyword and text similarity
    const keywordSimilarity = this.calculateKeywordSimilarity(
      post1.keywords,
      post2.keywords
    );
    const textSimilarity = this.calculateTextSimilarity(post1.body, post2.body);

    return (keywordSimilarity + textSimilarity) / 2;
  }

  /**
   * Calculate keyword similarity using Jaccard index
   */
  private calculateKeywordSimilarity(
    keywords1: string[],
    keywords2: string[]
  ): number {
    const set1 = new Set(keywords1.map((k) => k.toLowerCase()));
    const set2 = new Set(keywords2.map((k) => k.toLowerCase()));

    const intersection = new Set([...set1].filter((k) => set2.has(k)));
    const union = new Set([...set1, ...set2]);

    return union.size > 0 ? intersection.size / union.size : 0;
  }

  /**
   * Simple text similarity using common words
   */
  private calculateTextSimilarity(text1: string, text2: string): number {
    const words1 = new Set(
      text1
        .toLowerCase()
        .split(/\s+/)
        .filter((w) => w.length > 3)
    );
    const words2 = new Set(
      text2
        .toLowerCase()
        .split(/\s+/)
        .filter((w) => w.length > 3)
    );

    const intersection = new Set([...words1].filter((w) => words2.has(w)));
    const union = new Set([...words1, ...words2]);

    return union.size > 0 ? intersection.size / union.size : 0;
  }

  /**
   * Enhance clusters with names, types, and metadata
   */
  private async enhanceClusters(
    postClusters: PostForClustering[][]
  ): Promise<EnhancedCluster[]> {
    const enhancedClusters: EnhancedCluster[] = [];

    for (let i = 0; i < postClusters.length; i++) {
      const posts = postClusters[i];

      if (!posts || posts.length <= 0) continue; // SKIP empty cluster, don't return

      const representativePost = this.findRepresentativePost(posts);

      const clusterName = await this.generateClusterName(posts);
      const clusterType = await this.classifyClusterType(posts);
      const clusterDescription = await this.generateClusterDescription(
        posts,
        clusterType
      );
      const centroid = this.calculateCentroid(posts);
      const metadata = this.buildClusterMetadata(posts);

      const enhancedCluster: EnhancedCluster = {
        id: i + 1,
        name: clusterName,
        type: clusterType,
        description: clusterDescription,
        centroid: JSON.stringify(centroid),
        category_id: this.getMajorityCategory(posts),
        member_count: posts.length,
        member_ids: posts.map((p) => p.id),
        representative_post_id: representativePost.id,
        metadata,
        created_at: new Date().toISOString(),
        last_recomputed_at: new Date().toISOString(),
      };

      enhancedClusters.push(enhancedCluster);
    }

    return enhancedClusters;
  }

  /**
   * Find the most representative post in a cluster
   */
  private findRepresentativePost(
    posts: PostForClustering[]
  ): PostForClustering {
    // Prioritize by classification confidence, then by body length
    return posts.reduce((best, current) => {
      if (current.classification_confidence > best.classification_confidence) {
        return current;
      }
      if (
        current.classification_confidence === best.classification_confidence
      ) {
        return current.body.length > best.body.length ? current : best;
      }
      return best;
    });
  }

  /**
   * Generate cluster name using LLM
   */
  private async generateClusterName(
    posts: PostForClustering[]
  ): Promise<string> {
    const sampledBodies = posts
      .slice(0, 3)
      .map((p) => p.body)
      .join("\n---\n");

    const prompt = `Generate a concise cluster name (5-10 words) for these related posts:

${sampledBodies}

Requirements:
- Use title-case
- Capture the core issue or theme
- Be descriptive but concise
- No punctuation except hyphens

Cluster name:`;

    const llmResult = await llmClient.generateCompletion(prompt, 30);

    if (llmResult.success && llmResult.data) {
      return (llmResult.data as string).trim().replace(/[^a-zA-Z0-9\s-]/g, "");
    }

    // Fallback to keyword-based name
    const keywords = posts.flatMap((p) => p.keywords);
    const topKeywords = this.getTopKeywords(keywords, 3);
    return (
      topKeywords.join(" ").replace(/[^a-zA-Z0-9\s-]/g, "") || "Unnamed Cluster"
    );
  }

  /**
   * Classify cluster as problem or solution
   */
  private async classifyClusterType(
    posts: PostForClustering[]
  ): Promise<"problem" | "solution"> {
    const sampledBodies = posts
      .slice(0, 3)
      .map((p) => p.body)
      .join("\n---\n");

    const prompt = `Classify this cluster of posts as either "problem" or "solution":

${sampledBodies}

Instructions:
- "problem": If posts describe pain points, issues, complaints, or ask for help
- "solution": If posts propose features, approaches, fixes, or actionable ideas

Most posts in this cluster represent a:`;

    const llmResult = await llmClient.generateCompletion(prompt, 10);

    if (llmResult.success && llmResult.data) {
      const classification = (llmResult.data as string).toLowerCase().trim();
      if (classification.includes("solution")) {
        return "solution";
      }
    }

    return "problem"; // Default to problem
  }

  /**
   * Generate cluster description
   */
  private async generateClusterDescription(
    posts: PostForClustering[],
    type: "problem" | "solution"
  ): Promise<string> {
    const sampledBodies = posts
      .slice(0, 2)
      .map((p) => p.body)
      .join("\n---\n");
    const typeContext =
      type === "problem" ? "issue or pain point" : "solution or approach";

    const prompt = `Generate a 1-2 sentence description for this ${typeContext} cluster:

${sampledBodies}

The description should explain why this matters and its business impact. Keep it professional and concise.

Description:`;

    const llmResult = await llmClient.generateCompletion(prompt, 100);

    if (llmResult.success && llmResult.data) {
      return (llmResult.data as string).trim();
    }

    return `A cluster representing a ${type} affecting multiple users across different platforms.`;
  }

  /**
   * Calculate centroid from post embeddings
   */
  private calculateCentroid(posts: PostForClustering[]): number[] {
    const embeddings = posts
      .filter((p) => p.embedding)
      .map((p) => p.embedding as number[]);

    if (embeddings.length === 0) {
      return Array(1536).fill(0); // Default embedding size
    }

    return embeddingsService.calculateCentroid(embeddings);
  }

  /**
   * Build cluster metadata
   */
  private buildClusterMetadata(posts: PostForClustering[]): ClusterMetadata {
    // Aggregate keywords (top 6)
    const allKeywords = posts.flatMap((p) => p.keywords);
    const keywords = this.getTopKeywords(allKeywords, 6);

    // Count sources
    const sources: { [source: string]: number } = {};
    posts.forEach((p) => {
      sources[p.source] = (sources[p.source] || 0) + 1;
    });

    // Calculate aggregated confidence
    const aggregated_confidence =
      posts.reduce((sum, p) => sum + p.classification_confidence, 0) /
      posts.length;

    // Find date range
    const dates = posts
      .filter((p) => p.created_at)
      .map((p) => new Date(p.created_at!).getTime())
      .sort();

    const first_seen =
      dates.length > 0
        ? new Date(Math.min(...dates)).toISOString()
        : new Date().toISOString();
    const last_seen =
      dates.length > 0
        ? new Date(Math.max(...dates)).toISOString()
        : new Date().toISOString();

    return {
      keywords,
      sources,
      aggregated_confidence,
      first_seen,
      last_seen,
    };
  }

  /**
   * Get top keywords by frequency
   */
  private getTopKeywords(keywords: string[], limit: number): string[] {
    const counts = new Map<string, number>();

    keywords.forEach((keyword) => {
      const lower = keyword.toLowerCase();
      counts.set(lower, (counts.get(lower) || 0) + 1);
    });

    return Array.from(counts.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, limit)
      .map(([keyword]) => keyword);
  }

  /**
   * Get majority category among cluster members
   */
  private getMajorityCategory(posts: PostForClustering[]): number | null {
    const counts = new Map<number, number>();

    posts.forEach((post) => {
      if (post.category_id !== null) {
        counts.set(post.category_id, (counts.get(post.category_id) || 0) + 1);
      }
    });

    if (counts.size === 0) return null;

    const sortedEntries = Array.from(counts.entries()).sort(
      (a, b) => b[1] - a[1]
    );
    return sortedEntries?.[0]?.[0] ?? null;
  }

  /**
   * Merge clusters with similarity >= 0.85
   */
  private async mergeSimilarClusters(
    clusters: EnhancedCluster[]
  ): Promise<EnhancedCluster[]> {
    const merged = new Set<number>();
    const result: EnhancedCluster[] = [];

    for (let i = 0; i < clusters.length; i++) {
      if (merged.has(i)) continue;

      const cluster1 = clusters[i] as EnhancedCluster;
      const mergeCandidates: EnhancedCluster[] = [cluster1];

      for (let j = i + 1; j < clusters.length; j++) {
        if (merged.has(j)) continue;

        const cluster2 = clusters[j] as EnhancedCluster;
        const similarity: number = this.calculateClusterSimilarity(
          cluster1,
          cluster2
        );

        if (similarity >= 0.85) {
          mergeCandidates.push(cluster2);
          merged.add(j);
        }
      }

      if (mergeCandidates.length > 1) {
        const mergedCluster: EnhancedCluster =
          await this.mergeClusters(mergeCandidates);
        result.push(mergedCluster);
      } else {
        result.push(cluster1);
      }
    }

    return result;
  }
  /**
   * Calculate similarity between two clusters using centroids
   */
  private calculateClusterSimilarity(
    cluster1: EnhancedCluster,
    cluster2: EnhancedCluster
  ): number {
    try {
      const centroid1 = JSON.parse(cluster1.centroid) as number[];
      const centroid2 = JSON.parse(cluster2.centroid) as number[];

      return embeddingsService.calculateCosineSimilarity(centroid1, centroid2);
    } catch (error) {
      return 0;
    }
  }

  /**
   * Merge multiple clusters into one
   */
  private async mergeClusters(
    clusters: EnhancedCluster[]
  ): Promise<EnhancedCluster> {
    const mainCluster = clusters[0];
    const allMemberIds = clusters.flatMap((c) => c.member_ids);
    const allMemberCounts = clusters.reduce(
      (sum, c) => sum + c.member_count,
      0
    );

    // Merge centroids
    const centroids = clusters.map((c) => JSON.parse(c.centroid) as number[]);
    const mergedCentroid = embeddingsService.calculateCentroid(centroids);

    // Merge metadata
    const allKeywords = clusters.flatMap((c) => c.metadata.keywords);
    const mergedKeywords = this.getTopKeywords(allKeywords, 6);

    const mergedSources: { [source: string]: number } = {};
    clusters.forEach((c) => {
      Object.entries(c.metadata.sources).forEach(([source, count]) => {
        mergedSources[source] = (mergedSources[source] || 0) + count;
      });
    });

    const aggregatedConfidence =
      clusters.reduce(
        (sum, c) => sum + c.metadata.aggregated_confidence * c.member_count,
        0
      ) / allMemberCounts;

    const allFirstSeen = clusters.map((c) =>
      new Date(c.metadata.first_seen).getTime()
    );
    const allLastSeen = clusters.map((c) =>
      new Date(c.metadata.last_seen).getTime()
    );

    return {
      ...mainCluster,
      member_count: allMemberCounts,
      member_ids: allMemberIds,
      centroid: JSON.stringify(mergedCentroid),
      metadata: {
        keywords: mergedKeywords,
        sources: mergedSources,
        aggregated_confidence: aggregatedConfidence,
        first_seen: new Date(Math.min(...allFirstSeen)).toISOString(),
        last_seen: new Date(Math.max(...allLastSeen)).toISOString(),
      },
      last_recomputed_at: new Date().toISOString(),
    };
  }
}
